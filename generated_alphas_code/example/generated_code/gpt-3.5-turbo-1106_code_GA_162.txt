df['rank_stddev_open_10'] = df['open'].rolling(window=10).std().rank(axis=1, pct=True)
# Rank the volume and calculate the correlation with high prices over a 10-day period
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['correlation_rank_volume_high_10'] = df['rank_volume'].rolling(window=10).corr(df['high'])
# Mutate the alpha factor according to the given formula
df['mutated_alpha'] = df['correlation_rank_volume_high_10'] * (-1 * df['rank_stddev_open_10'])
# Rank the minimum opening prices over a 5-day period
df['rank_ts_min_open_5'] = df['open'].rolling(window=5).min().rank(axis=1, pct=True)
# Calculate the final alpha factor
df['alpha'] = df['mutated_alpha'].rolling(window=15).min() + df['rank_ts_min_open_5']
