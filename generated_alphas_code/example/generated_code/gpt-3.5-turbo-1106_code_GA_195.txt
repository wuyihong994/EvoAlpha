

# Mutated Alpha: rank(ts_min(rank(df['open']), 5)) * signedpower(rank(sum(rank(df['volume']), 5)), 0.5) * correlation(rank(rank(correlation(rank(df['volume']), rank(df['high']), 3)), 5) - delay(stddev(df['high'], 15), 5)
df['rank_open'] = df['open'].rank(axis=1, pct=True)
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['rank_high'] = df['high'].rank(axis=1, pct=True)
df['rank_ts_min_open'] = df['rank_open'].rolling(window=5).min()
df['rank_sum_volume'] = df['rank_volume'].rolling(window=5).sum()
df['rank_corr_volume_high'] = df['rank_volume'].rolling(window=3).corr(df['rank_high'])
df['rank_corr_corr_volume_high'] = df['rank_corr_volume_high'].rank(axis=1, pct=True)
df['signedpower_rank_sum_volume'] = df['rank_sum_volume'] ** 0.5
df['correlation_delay_stddev'] = df['rank_corr_corr_volume_high'] - df['high'].rolling(window=15).std().shift(5)
df['alpha'] = df['rank_ts_min_open'] * df['signedpower_rank_sum_volume'] * df['correlation_delay_stddev']
