df['rank_high_5'] = df['high'].rolling(window=5).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
# Rank the volume over the past 5 days
df['rank_volume_5'] = df['volume'].rolling(window=5).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
# Compute the product of the ranks
df['product_rank_volume_5'] = df['rank_volume_5'] * df['rank_volume_5']
# Compute the correlation between high and volume over the past 10 days
df['correlation_high_volume_10'] = df['high'].rolling(window=10).corr(df['volume'])
# Calculate daily returns
df['returns'] = df['close'].pct_change()
# Calculate the alpha factor
df['alpha'] = ((-1 * df['rank_high_5']) + df['product_rank_volume_5']) * df['correlation_high_volume_10'] * df['returns']
