df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the time-series minimum of the ranked open prices over the past 5 days
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
# Rank the results
df['rank_ts_min_rank_open_5'] = df['ts_min_rank_open_5'].rank(axis=1, pct=True)
# Calculate the rank of the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the sum of the ranked volume over the past 5 days
df['sum_rank_volume_5'] = df['rank_volume'].rolling(window=5).sum()
# Rank the results
df['rank_sum_rank_volume_5'] = df['sum_rank_volume_5'].rank(axis=1, pct=True)
# Calculate the SignedPower of the ranked sum of volume
df['signedpower_rank_sum_volume_5'] = df['rank_sum_rank_volume_5'] ** 0.5
# Calculate the correlation of volume and ranked high prices over the past 10 days
df['correlation_volume_rank_high_10'] = df['volume'].rolling(window=10).corr(df['high'])
# Rank the results
df['rank_correlation_volume_rank_high_10'] = df['correlation_volume_rank_high_10'].rank(axis=1, pct=True)
# Calculate the time-series maximum of the ranked correlation over the past 5 days
df['ts_max_rank_correlation_volume_rank_high_10_5'] = df['rank_correlation_volume_rank_high_10'].rolling(window=5).max()
# Rank the results
df['rank_ts_max_rank_correlation_volume_rank_high_10_5'] = df['ts_max_rank_correlation_volume_rank_high_10_5'].rank(axis=1, pct=True)
# Calculate the standard deviation of high prices over the past 10 days
df['stddev_high_10'] = df['high'].rolling(window=10).std()
# Calculate the final alpha
df['alpha'] = -1 * df['rank_ts_min_rank_open_5'] * df['signedpower_rank_sum_volume_5'] * df['rank_ts_max_rank_correlation_volume_rank_high_10_5'] / df['stddev_high_10']
