df['stddev_open_10'] = df['open'].rolling(window=10).std()
# Calculate the rank of the standard deviation
df['rank_stddev_open_10'] = df['stddev_open_10'].rank(axis=1, pct=True)
# Calculate the rank of the volume and high prices
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['rank_high'] = df['high'].rank(axis=1, pct=True)
# Calculate the correlation of the ranked volume and high prices over the past 3 days
df['correlation_volume_high_3'] = df['rank_volume'].rolling(window=3).corr(df['rank_high'])
# Calculate the product of the negative rank of the standard deviation and the correlation, and then take the time-series minimum over the past 3 days
df['alpha'] = df['rank_stddev_open_10'] * -1 * df['correlation_volume_high_3'].rolling(window=3).min()
