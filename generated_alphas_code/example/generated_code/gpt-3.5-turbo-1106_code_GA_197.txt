# Calculate the time-series min rank of the open prices over the past 5 days
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
# Calculate the rank of the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the product of the rank of the volume over the past 5 days
df['product_rank_volume_5'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)
# Calculate the rank of the open prices again
df['rank_rank_open'] = df['rank_open'].rank(axis=1, pct=True)
# Calculate the rank of the product of the rank of the volume over the past 5 days and the rank of the open prices over the past 5 days
df['rank_product_rank_volume_5'] = df['product_rank_volume_5'] * df['rank_rank_open']
# Calculate the rank of the rank of the rank of the open prices
df['rank_rank_rank_open'] = df['rank_open'].rank(axis=1, pct=True)
# Calculate the division of the rank of the product of the rank of the volume over the past 5 days and the rank of the open prices over the past 5 days by the rank of the rank of the rank of the open prices
df['division'] = df['rank_product_rank_volume_5'] / df['rank_rank_rank_open']
# Calculate the daily returns
df['returns'] = df['close'].pct_change()
# Calculate the 5-day average volume
df['adv5'] = df['volume'].rolling(window=5).mean()
# Calculate the final alpha
df['alpha'] = -1 * df['division'] / (df['adv5'] * df['returns'])
