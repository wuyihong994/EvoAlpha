df['rank_stddev_high_5'] = df['high'].rolling(window=5).std().rank(axis=1, pct=True)
# Calculate the correlation between high and the closing price 3 days ago over the past 10 days
df['correlation_high_close_delayed'] = df['high'].rolling(window=10).corr(df['close'].shift(3))
# Scale the rank of volume by 2
df['scaled_rank_volume'] = df['volume'].rank(axis=1, pct=True) * 2
# Calculate the absolute rank of the opening prices
df['abs_rank_open'] = abs(df['open'].rank(axis=1, pct=True))
# Compute the signed power of the rank of the standard deviation of high prices and subtract the product of correlation, scaled rank of volume, and absolute rank of opening prices
df['alpha'] = (df['rank_stddev_high_5'] ** 0.5) - (df['correlation_high_close_delayed'] * df['scaled_rank_volume'] * df['abs_rank_open'])
