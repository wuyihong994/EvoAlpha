df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Compute the time-series minimum rank of the open prices over the past 5 days
df['ts_argmin_rank_open_5'] = df['rank_open'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True)
# Rank the volume and calculate the product over the past 5 days
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['product_rank_volume_5'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)
# Rank the results and calculate the final alpha
df['alpha'] = -1 * df['ts_argmin_rank_open_5'].rank(axis=1, pct=True) + df['product_rank_volume_5'].rank(axis=1, pct=True) + df['rank_open'].rank(axis=1, pct=True).rank(axis=1, pct=True)
