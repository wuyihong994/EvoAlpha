df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the time-series min over the past 5 days and rank the result
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min().rank(axis=1, pct=True)
# Rank the volume and calculate the product over the past 5 days, then rank the result
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['product_rank_volume_5'] = (df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)).rank(axis=1, pct=True)
# Combine the ranks and rank the result
df['rank_combined'] = df['ts_min_rank_open_5'] + df['product_rank_volume_5']
df['rank_rank_combined'] = df['rank_combined'].rank(axis=1, pct=True)
# Rank the open prices multiple times
df['rank_rank_open'] = df['open'].rank(axis=1, pct=True).rank(axis=1, pct=True)
# Calculate the correlation between high and volume over the past 10 days and rank the result
df['correlation_high_volume_10'] = df['high'].rolling(window=10).corr(df['volume']).rank(axis=1, pct=True)
# Calculate the delta of close prices over 3 days and apply signedpower
df['delta_close_3'] = df['close'].diff(3)
df['signedpower_delta_close_2'] = (df['delta_close_3'] ** 2)
# Combine all factors and calculate the alpha
df['alpha'] = ((-1 * df['rank_rank_combined'] + df['rank_rank_open']) * df['correlation_high_volume_10'] + df['signedpower_delta_close_2'])
