df['rank_high_10'] = df['high'].rolling(window=10).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
# Calculate the standard deviation of the ranked high prices over the past 10 days
df['stddev_rank_high_10'] = df['rank_high_10'].rolling(window=10).std()
# Calculate the rank of the open prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the correlation between the ranked open prices and the volume over the past 10 days
df['correlation_open_volume_10'] = df['rank_open'].rolling(window=10).corr(df['volume'])
# Calculate the sum of the product of the ranked standard deviation of high prices and the correlation between the ranked open prices and volume over the past 15 days
df['alpha'] = ((-1 * df['stddev_rank_high_10'].rank(axis=1, pct=True)) * df['correlation_open_volume_10']).rolling(window=15).sum().rank(axis=1, pct=True)
