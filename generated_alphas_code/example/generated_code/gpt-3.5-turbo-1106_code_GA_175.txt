df['rank_stddev_open_10'] = df['open'].rolling(window=10).std().rank(axis=1, pct=True)
# Rank the volume and calculate the correlation with high prices over a 10-day period
df['correlation_rank_volume_high_10'] = df['volume'].rank(axis=1, pct=True).rolling(window=10).corr(df['high'])
# Multiply the negative rank of the standard deviation by the correlation and calculate the time-series minimum over the past 15 days
df['alpha'] = df['rank_stddev_open_10'].apply(lambda x: -1 * x) * df['correlation_rank_volume_high_10']
df['alpha'] = df['alpha'].rolling(window=15).min()
