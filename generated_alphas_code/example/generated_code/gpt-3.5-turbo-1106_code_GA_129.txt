df['returns'] = df['close'].pct_change()
# Calculate the cross-sectional rank of the 'open' prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the cross-sectional rank of the 'returns'
df['rank_returns'] = df['returns'].rank(axis=1, pct=True)
# Calculate the 5-day time-series correlation of the rank of 'open' and the rank of 'returns'
df['correlation_rank_open_returns_5'] = df['rank_open'].rolling(window=5).corr(df['rank_returns'])
# Rank the correlation results
df['rank_correlation'] = df['correlation_rank_open_returns_5'].rank(axis=1, pct=True)
# Find the index of the minimum value over the past 5 days and multiply by -1
df['alpha'] = -1 * df['rank_correlation'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True)
