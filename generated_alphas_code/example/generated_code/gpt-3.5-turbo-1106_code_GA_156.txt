df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Compute the rank of the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Compute the time-series min over the past 5 days of the rank of the open price
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
# Compute the product of the rank of volume over the past 5 days
df['product_rank_volume_5'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)
# Compute the first rank
df['rank_ts_min_rank_open_5'] = df['ts_min_rank_open_5'].rank(axis=1, pct=True)
# Compute the second rank
df['rank_product_rank_volume_5'] = df['product_rank_volume_5'].rank(axis=1, pct=True)
# Compute the delta of the rank of open price over the past 10 days
df['delta_rank_open_10'] = df['rank_open'].diff(10)
# Compute the alpha
df['alpha'] = -1 * (df['rank_ts_min_rank_open_5'] + df['rank_product_rank_volume_5']).rank(axis=1, pct=True) + df['rank_product_rank_volume_5'] + df['delta_rank_open_10'].rank(axis=1, pct=True)
