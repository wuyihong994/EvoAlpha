

# Calculate the rank of open prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Delay the rank of open prices by 5 days
df['delay_rank_open_5'] = df['rank_open'].shift(5)
# Calculate the rank of the delayed rank of open prices
df['rank_delay_rank_open_5'] = df['delay_rank_open_5'].rank(axis=1, pct=True)
# Calculate the time-series minimum of the ranked delayed rank of open prices over the past 5 days
df['ts_argmin'] = df['rank_delay_rank_open_5'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True)
# Rank the results and multiply by -1
df['alpha_part1'] = (-1 * df['ts_argmin'].rank(axis=1, pct=True))
# Calculate the rank of open prices and rank the results twice
df['rank_open_twice'] = df['rank_open'].rank(axis=1, pct=True).rank(axis=1, pct=True)
# Add the double-ranked open prices to alpha_part1
df['alpha_part2'] = df['alpha_part1'] + df['rank_open_twice']
# Calculate the rank of volume and vwap
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['rank_vwap'] = df['vwap'].rank(axis=1, pct=True)
# Calculate the correlation of the ranked volume and vwap over the past 5 days
df['correlation_rank_volume_vwap_5'] = df['rank_volume'].rolling(window=5).corr(df['rank_vwap'])
# Calculate the rank of the correlation results
df['rank_correlation'] = df['correlation_rank_volume_vwap_5'].rank(axis=1, pct=True)
# Calculate the time-series maximum of the ranked correlation results over the past 5 days and multiply by -1
df['ts_max_rank_correlation_5'] = (-1 * df['rank_correlation'].rolling(window=5).max())
# Add the max-ranked correlation to alpha_part2
df['alpha'] = df['alpha_part2'] + df['ts_max_rank_correlation_5']
