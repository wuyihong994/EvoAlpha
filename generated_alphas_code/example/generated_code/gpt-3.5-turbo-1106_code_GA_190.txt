rank_high = df['high'].rolling(window=5, min_periods=1).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
rank_volume = df['volume'].rolling(window=5, min_periods=1).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
# Compute the covariance of the rank of high prices and the rank of volume over a 5-day period
cov_rank_high_volume = (rank_high.rolling(window=5, min_periods=1).cov(rank_volume)).rank(axis=1, pct=True)
# Calculate the standard deviation of high prices over a 10-day period
stddev_high_10 = df['high'].rolling(window=10, min_periods=1).std()
# Compute the final alpha
df['alpha'] = (-1 * cov_rank_high_volume) * stddev_high_10
