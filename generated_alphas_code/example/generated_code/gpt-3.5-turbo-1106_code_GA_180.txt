df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the time-series min over the past 5 days and rank the result
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min().rank(axis=1, pct=True)
# Calculate the rank of the 'volume' and 'high' prices
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['rank_high'] = df['high'].rank(axis=1, pct=True)
# Calculate the correlation of the ranked 'volume' and 'high' prices over the past 3 days and rank the result
df['correlation_rank_volume_high_3'] = df['rank_volume'].rolling(window=3).corr(df['rank_high']).rank(axis=1, pct=True)
# Calculate the sum of the ranked correlation over the past 3 days
df['sum_rank_correlation_volume_high_3'] = df['correlation_rank_volume_high_3'].rolling(window=3).sum()
# Calculate the double rank of the time-series min of the rank of 'open' over the past 5 days
df['double_rank_ts_min_rank_open_5'] = df['ts_min_rank_open_5'].rank(axis=1, pct=True).rank(axis=1, pct=True)
# Calculate the final alpha factor
df['alpha'] = df['double_rank_ts_min_rank_open_5'].rolling(window=2).mean().shift(2)
