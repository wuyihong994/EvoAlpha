df['returns'] = df['close'].pct_change()
# Calculate the cross-sectional rank of returns and high over the past 3 days
rank_returns = df['returns'].rank(axis=1, pct=True)
rank_high = df['high'].rank(axis=1, pct=True)
# Calculate the cross-sectional correlation of the ranked returns and high over the past 3 days
correlation_rank_returns_high = rank_returns.rolling(window=3).corr(rank_high)
# Sum the rank of the correlation over the past 3 days
sum_rank_correlation = correlation_rank_returns_high.rolling(window=3).sum()
# Calculate the time-series rank of the opening price over the past 10 days
ts_rank_open_10 = df['open'].rolling(window=10).apply(lambda x: (x[-1] - min(x)) / (max(x) - min(x)), raw=True)
# Combine the elements to compute the final alpha
df['alpha'] = (-1 * sum_rank_correlation) * ts_rank_open_10
