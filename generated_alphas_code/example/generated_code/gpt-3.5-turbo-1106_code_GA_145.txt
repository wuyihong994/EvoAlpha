df['delta_close_5'] = abs(df['close'] - df['close'].shift(5))
# Calculate the rank of the closing price
df['rank_close'] = df['close'].rank(axis=1, pct=True)
# Calculate the correlation between volume and rank of the closing price over the past 5 days
df['correlation_volume_rank_close_5'] = df['volume'].rolling(window=5).corr(df['rank_close'])
# Calculate the time-series min of the correlation over the past 10 days
df['ts_min_corr_vol_rank_close_10'] = df['correlation_volume_rank_close_5'].rolling(window=10).min()
# Calculate the alpha factor
df['alpha'] = df['delta_close_5'] * df['ts_min_corr_vol_rank_close_10']
