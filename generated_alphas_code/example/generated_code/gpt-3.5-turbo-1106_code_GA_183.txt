df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the time-series minimum of the rank of open price over the past 5 days
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
# Calculate the rank of the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the sum of the rank of volume over the past 5 days
df['sum_rank_volume_5'] = df['rank_volume'].rolling(window=5).sum()
# Calculate the signed power of the sum of rank of volume
df['signedpower_sum_rank_volume_5'] = df['sum_rank_volume_5'].rank(axis=1, pct=True) ** 0.5
# Calculate the rank of the open price three times
df['rank_rank_rank_open'] = df['rank_open'].rank(axis=1, pct=True).rank(axis=1, pct=True).rank(axis=1, pct=True)
# Calculate the advanced rolling mean of the close price over the past 5 days
df['adv5'] = df['close'].rolling(window=5).mean()
# Calculate the alpha factor
df['alpha'] = -1 * df['ts_min_rank_open_5'].rank(axis=1, pct=True) * df['signedpower_sum_rank_volume_5'] * df['rank_rank_rank_open'] / df['adv5']
