

# Calculate the standard deviation of the closing price over a 10-day period
df['stddev_close_10'] = df['close'].rolling(window=10).std()
# Rank the standard deviation over the past 5 days
df['rank_stddev_close_10'] = df['stddev_close_10'].rolling(window=5).apply(lambda x: pd.Series(x).rank(pct=True).iloc[-1], raw=True)
# Calculate the time-series max over the past 5 days and multiply by -1
df['ts_max_rank_stddev_close_10'] = -1 * df['rank_stddev_close_10'].rolling(window=5).max()
# Calculate the time-serial correlation of open and volume for the past 10 days
df['correlation_open_volume_10'] = df['open'].rolling(window=10).corr(df['volume'])
# Rank the open price
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Create the alpha feature
df['alpha'] = df['ts_max_rank_stddev_close_10'] * df['correlation_open_volume_10'] * df['rank_open']
