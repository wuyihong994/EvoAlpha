

# Mutate Alpha: ts_min(((-1 * rank(stddev(open, 10))) * correlation(rank(volume), high, 10) - stddev(open, 10)), 5)
# Calculate the standard deviation of opening prices over a 10-day period
df['stddev_open_10'] = df['open'].rolling(window=10).std()
# Rank the standard deviation of opening prices
df['rank_stddev_open_10'] = df['stddev_open_10'].rank(axis=1, pct=True)
# Rank the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the correlation between the ranked volume and high prices over a 10-day period
df['correlation_rank_volume_high_10'] = df['rank_volume'].rolling(window=10).corr(df['high'])
# Calculate the first part of the alpha expression
df['alpha_part1'] = (-1 * df['rank_stddev_open_10']) * df['correlation_rank_volume_high_10'] - df['stddev_open_10']
# Calculate the ts_min of the alpha expression over a 5-day period
df['alpha'] = df['alpha_part1'].rolling(window=5).min()
