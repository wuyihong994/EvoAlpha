df['rank_high'] = df['high'].rank(axis=1, pct=True)
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the covariance of the ranked high and volume features over the past 5 days
df['cov_rank_high_volume_5'] = df['rank_high'].rolling(window=5).cov(df['rank_volume'])
# Calculate the delta of the volume feature over the past 10 days
df['delta_volume_10'] = df['volume'] - df['volume'].shift(10)
# Calculate the absolute value of the rank covariance and take the logarithm
df['log_abs_rank_cov'] = np.log(np.abs(df['cov_rank_high_volume_5']))
# Create the alpha feature based on the given formula
df['alpha'] = df['log_abs_rank_cov'] * df['delta_volume_10']
