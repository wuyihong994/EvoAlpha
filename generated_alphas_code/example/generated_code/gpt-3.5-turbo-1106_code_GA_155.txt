

# New Mutate Alpha: ts_min((-1 * rank(stddev(high, 10))) * correlation(rank(volume), high, 10), 15) * rank(open) / abs(delta(volume, 5))
# Calculate the standard deviation of high prices over a 10-day period
df['stddev_high_10'] = df['high'].rolling(window=10).std()
# Rank the standard deviation
df['rank_stddev_high_10'] = df['stddev_high_10'].rank(axis=1, pct=True)
# Rank the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the correlation between the ranked volume and high prices over a 10-day period
df['correlation_rank_volume_high_10'] = df['rank_volume'].rolling(window=10).corr(df['high'])
# Calculate the first part of the alpha expression
df['part1_alpha'] = (-1 * df['rank_stddev_high_10']) * df['correlation_rank_volume_high_10']
# Calculate the ts_min over the past 15 days
df['ts_min_alpha'] = df['part1_alpha'].rolling(window=15).min()
# Rank the open prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the absolute delta of the volume over 5 days
df['abs_delta_volume_5'] = abs(df['volume'] - df['volume'].shift(5))
# Calculate the final alpha
df['alpha'] = df['ts_min_alpha'] * df['rank_open'] / df['abs_delta_volume_5']
