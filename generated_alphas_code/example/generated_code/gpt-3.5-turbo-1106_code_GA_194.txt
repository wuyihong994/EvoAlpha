# Calculate the time-series max of the ranked correlation over the past 5 days
df['ts_max_rank_correlation_high_rank_volume_10_5'] = df['rank_correlation_high_rank_volume_10'].rolling(window=5).max()
# Calculate the correlation between high and volume over the past 10 days
df['correlation_high_volume_10'] = df['high'].rolling(window=10).corr(df['volume'])
# Calculate the 5-day moving average of the volume
df['adv5'] = df['volume'].rolling(window=5).mean()
# Calculate the natural logarithm of adv5
df['log_adv5'] = np.log(df['adv5'])
# Calculate the alpha
df['alpha'] = -1 * df['ts_max_rank_correlation_high_rank_volume_10_5'] * df['correlation_high_volume_10'] * df['log_adv5']
