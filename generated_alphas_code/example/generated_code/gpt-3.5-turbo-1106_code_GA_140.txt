

# Mutate Alpha: (-1 * sum(rank(correlation(rank(high), rank(volume), 3)), 3)) * correlation(open, volume, 10) * rank(open) * returns
# Calculate the rank of high and volume
rank_high = df['high'].rank(axis=1, pct=True)
rank_volume = df['volume'].rank(axis=1, pct=True)
# Calculate the correlation of ranked high and volume over the past 3 days
correlation_ranked = rank_high.rolling(window=3).corr(rank_volume)
# Sum the ranks of the correlation over the past 3 days and multiply by -1
sum_rank_correlation = -1 * correlation_ranked.rolling(window=3).sum()
# Calculate the correlation of open and volume over the past 10 days
correlation_open_volume = df['open'].rolling(window=10).corr(df['volume'])
# Rank the open prices
rank_open = df['open'].rank(axis=1, pct=True)
# Calculate the daily returns
df['returns'] = df['close'].pct_change()
# Calculate the final alpha
df['alpha'] = sum_rank_correlation * correlation_open_volume * rank_open * df['returns']
