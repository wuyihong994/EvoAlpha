df['stddev_open_10'] = df['open'].rolling(window=10).std()
# Rank the standard deviation and multiply by -1
df['rank_stddev_open_10'] = df['stddev_open_10'].rank(axis=1, pct=True) * -1
# Rank the volume and calculate the correlation with high prices over a 10-day period
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['correlation_rank_volume_high_10'] = df['rank_volume'].rolling(window=10).corr(df['high'])
# Calculate the time-series min over the past 15 days of the product of the previous factors
df['alpha'] = df['correlation_rank_volume_high_10'].rolling(window=15).min()
