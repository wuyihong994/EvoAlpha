df['returns'] = df['close'].pct_change()
# Calculate the rank of the volume-weighted average price (vwap) and the returns
df['rank_vwap'] = df['vwap'].rank(axis=1, pct=True)
df['rank_returns'] = df['returns'].rank(axis=1, pct=True)
# Calculate the correlation between the ranked vwap and returns over the past 5 days
df['correlation_rank_vwap_returns_5'] = df['rank_vwap'].rolling(window=5).corr(df['rank_returns'])
# Calculate the rank of the correlation
df['rank_correlation'] = df['correlation_rank_vwap_returns_5'].rank(axis=1, pct=True)
# Calculate the time-series max over the past 5 days of the ranked correlation
df['ts_max_rank_correlation_5'] = df['rank_correlation'].rolling(window=5).max()
# Calculate the final alpha
df['alpha'] = -1 * df['ts_max_rank_correlation_5']
