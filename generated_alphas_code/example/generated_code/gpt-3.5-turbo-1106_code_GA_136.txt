

# Mutate Alpha: ((-1 * rank(delta(open, 15))) * correlation(open, volume, 10) * rank(open) * returns)
# Calculate the change in open price over a 15-day period
df['delta_open_15'] = df['open'] - df['open'].shift(15)
# Calculate the correlation between open price and volume over a 10-day period
df['correlation_open_volume_10'] = df['open'].rolling(window=10).corr(df['volume'])
# Rank the open price
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate daily returns
df['returns'] = df['close'].pct_change()
# Calculate the alpha
df['alpha'] = -1 * df['delta_open_15'].rank(axis=1, pct=True) * df['correlation_open_volume_10'] * df['rank_open'] * df['returns']
