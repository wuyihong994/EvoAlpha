df['returns'] = df['close'].pct_change()
# Delay the returns by 1 day and take the negative value
df['delayed_returns'] = df['returns'].shift(1) * -1
# Calculate the Ts_Rank of the delayed returns over the past 6 days and then rank the results over the past 5 days
df['Ts_Rank'] = df['delayed_returns'].rolling(window=5).apply(lambda x: (x[-1] > x).sum() / len(x), raw=True)
# Rank the correlation between open price and volume over the past 10 days
df['correlation_open_volume'] = df['open'].rolling(window=10).corr(df['volume'])
# Calculate the alpha factor
df['alpha'] = (0.73 * df['Ts_Rank'].rank(axis=1, pct=True)) + (-1 * df['correlation_open_volume']) + df['close']
