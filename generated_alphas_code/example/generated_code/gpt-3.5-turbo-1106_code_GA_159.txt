# Rank the standard deviation of high prices

# Rank open prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the 10-day returns
df['returns'] = df['close'].pct_change()

df['stddev_returns_10'] = df['returns'].rolling(window=10).std()
# Rank the standard deviation of returns
df['rank_stddev_returns_10'] = df['stddev_returns_10'].rank(axis=1, pct=True)
