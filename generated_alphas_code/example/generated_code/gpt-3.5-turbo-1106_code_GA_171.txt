df['min_rank_high'] = df['high'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True).rank(axis=1, pct=True)
# Calculate the product of the rank of volume over the past 5 days
df['product_rank_volume'] = (df['volume'].rank(axis=1, pct=True) ** 5)
# Calculate the sum of the ranks and then rank the result
df['sum_ranks'] = (df['min_rank_high'] + df['product_rank_volume']).rank(axis=1, pct=True)
# Calculate the correlation between high and volume over the past 10 days
df['correlation_high_volume_10'] = df['high'].rolling(window=10).corr(df['volume'])
# Calculate the daily returns
df['returns'] = df['close'].pct_change()
# Calculate the final alpha
df['alpha'] = -1 * df['sum_ranks'] * df['correlation_high_volume_10'] * df['returns']
