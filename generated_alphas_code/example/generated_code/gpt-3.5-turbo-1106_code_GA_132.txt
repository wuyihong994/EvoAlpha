

# Calculate the rank of open, high, and volume
df['rank_open'] = df['open'].rank(axis=1, pct=True)
df['rank_high'] = df['high'].rank(axis=1, pct=True)
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)

# Calculate the correlation of ranked open and ranked high over the past 3 days
df['correlation_rank_open_high'] = df['rank_open'].rolling(window=3).corr(df['rank_high'])

# Calculate the sum of the correlation ranks over the past 3 days
df['sum_correlation_rank'] = df['correlation_rank_open_high'].rolling(window=3).sum()

# Calculate the rank of the time-series argmax of the ranked open over the past 5 days
df['rank_ts_argmax_open'] = df['rank_open'].rolling(window=5).apply(lambda x: np.argmax(x), raw=True).rank(axis=1, pct=True)

# Calculate the product of the rank of volume over the past 5 days
df['product_rank_volume'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)

# Calculate the alpha feature
df['alpha'] = (-1 * df['sum_correlation_rank']) + df['rank_ts_argmax_open'] - df['product_rank_volume']
