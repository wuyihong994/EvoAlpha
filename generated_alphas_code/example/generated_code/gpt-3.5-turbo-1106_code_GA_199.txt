df['rank_stddev_open_10'] = df['open'].rolling(window=10).std().rank(axis=1, pct=True)
# Calculate the rank of the trading volume and the correlation with the high price over a 10-day period
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['correlation_rank_volume_high_10'] = df['rank_volume'].rolling(window=10).corr(df['high'])
# Multiply the negative rank of the standard deviation with the correlation and find the time-series max over the past 15 days
df['alpha'] = df['rank_stddev_open_10'] * -1 * df['correlation_rank_volume_high_10']
df['alpha'] = df['alpha'].rolling(window=15).max()
