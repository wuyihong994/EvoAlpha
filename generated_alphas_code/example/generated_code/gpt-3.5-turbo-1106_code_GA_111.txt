df['rank_stddev_high_10'] = df['high'].rolling(window=10).std().rank(axis=1, pct=True)
# Calculate the cross-sectional rank of the opening prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the cross-sectional rank of the correlation between the two previous ranks over a 5-day period
df['rank_corr'] = df['rank_stddev_high_10'].rolling(window=5).corr(df['rank_open']).rank(axis=1, pct=True)
# Calculate the time-series rank of the previous result over a 5-day period and multiply by -1
df['alpha'] = -1 * df['rank_corr'].rolling(window=5).apply(lambda x: np.argmax(x), raw=True).rank(axis=1, pct=True)
