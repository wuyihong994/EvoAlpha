df['returns'] = df['close'].pct_change()
# Calculate the standard deviation of returns over a 10-day period
df['stddev_returns_10'] = df['returns'].rolling(window=10).std()
# Compute the Ts_ArgMax (index of the maximum value) over the past 5 days
df['Ts_ArgMax_returns_10'] = df['returns'].rolling(window=5).apply(lambda x: np.argmax(x), raw=True)
# Rank the results and subtract 0.5
df['alpha'] = df['Ts_ArgMax_returns_10'].rank(axis=1, pct=True) - 0.5
