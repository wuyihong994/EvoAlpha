df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the time-series min over the past 5 days of the ranked opening price
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
# Calculate the rank of the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the sum of the ranked volume over the past 5 days
df['sum_rank_volume_5'] = df['rank_volume'].rolling(window=5).sum()
# Calculate the signed power of the sum of the ranked volume
df['signedpower_sum_rank_volume_5'] = df['sum_rank_volume_5'].rank(axis=1, pct=True) ** 0.5
# Calculate the rank of the high price
df['rank_high'] = df['high'].rank(axis=1, pct=True)
# Calculate the correlation between volume and the ranked high price over the past 10 days
df['correlation_volume_rank_high_10'] = df['volume'].rolling(window=10).corr(df['rank_high'])
# Calculate the time-series max over the past 5 days of the correlation
df['ts_max_correlation_volume_rank_high_10_5'] = df['correlation_volume_rank_high_10'].rolling(window=5).max()
# Calculate the final alpha
df['alpha'] = -1 * df['ts_min_rank_open_5'].rank(axis=1, pct=True) * df['signedpower_sum_rank_volume_5'] * df['ts_max_correlation_volume_rank_high_10_5'].rank(axis=1, pct=True)
