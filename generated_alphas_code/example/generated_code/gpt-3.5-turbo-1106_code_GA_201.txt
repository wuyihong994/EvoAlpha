df['rank_open'] = df['open'].rank(axis=1, pct=True)
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['rank_high'] = df['high'].rank(axis=1, pct=True)
df['ts_min_rank_open'] = df['rank_open'].rolling(window=5).min()
df['rank_product_volume'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)
df['rank_sum_volume'] = df['rank_volume'].rolling(window=5).sum()
df['rank_corr_volume_high'] = df['volume'].rolling(window=10).corr(df['rank_high'])
df['alpha'] = (-1 * df['ts_min_rank_open'].rank(axis=1, pct=True) + df['rank_product_volume'].rank(axis=1, pct=True)) * (df['rank_sum_volume'].rank(axis=1, pct=True) ** 0.5) * df['rank_corr_volume_high'].rolling(window=5).max() / (df['high'].rolling(window=10).std() + df['rank_open'].rank(axis=1, pct=True))
