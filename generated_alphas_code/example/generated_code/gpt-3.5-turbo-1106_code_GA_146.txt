df['returns'] = df['close'].pct_change()
# Rank the returns
df['rank_returns'] = df['returns'].rank(axis=1, pct=True)
# Rank the high prices
df['rank_high'] = df['high'].rank(axis=1, pct=True)
# Calculate the correlation of the ranked returns and high prices over the past 3 days
df['correlation_returns_high_3'] = df['rank_returns'].rolling(window=3).corr(df['rank_high'])
# Rank the correlation results
df['rank_correlation_returns_high_3'] = df['correlation_returns_high_3'].rank(axis=1, pct=True)
# Find the maximum rank of the correlation over the past 3 days
df['ts_max_rank_correlation_returns_high_3'] = df['rank_correlation_returns_high_3'].rolling(window=3).max()
# Rank the results and take the minimum over the past 3 days, then multiply by -1
df['alpha'] = -1 * df['ts_max_rank_correlation_returns_high_3'].rolling(window=3).min().rank(axis=1, pct=True)
