df['rank_open'] = df['open'].rank(axis=1, pct=True)
df['ts_min_rank_open_5'] = df['rank_open'].rolling(window=5).min()
df['rank_ts_min_rank_open_5'] = df['ts_min_rank_open_5'].rank(axis=1, pct=True)

# Calculate the rank of open and then rank it again
df['rank_rank_open'] = df['rank_open'].rank(axis=1, pct=True)

# Calculate the product of the rank of volume over the past 5 days and then rank it
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
df['product_rank_volume_5'] = df['rank_volume'].rolling(window=5).apply(lambda x: np.prod(x), raw=True)
df['rank_product_rank_volume_5'] = df['product_rank_volume_5'].rank(axis=1, pct=True)

# Calculate the final alpha
df['alpha'] = df['rank_ts_min_rank_open_5'] + df['rank_rank_open'] - df['rank_product_rank_volume_5']
