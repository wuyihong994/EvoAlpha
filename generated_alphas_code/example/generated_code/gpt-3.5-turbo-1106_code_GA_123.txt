df['adv5'] = (df['volume'] * df['vwap']).rolling(window=5).mean()
# Calculate the correlation between high and the rank of vwap over the past 5 days
df['correlation'] = df['high'].rolling(window=5).corr(df['vwap'].rank(axis=1, pct=True))
# Rank the correlation results
df['rank_corr'] = df['correlation'].rank(axis=1, pct=True)
# Calculate the time-series max over the past 5 days of the ranked correlation
df['ts_max_rank_corr_5'] = df['rank_corr'].rolling(window=5).max()
# Rank the standard deviation of adv5 over the past 10 days
df['rank_stddev_adv5_10'] = df['adv5'].rolling(window=10).std().rank(axis=1, pct=True)
# Calculate the final alpha
df['alpha'] = -1 * df['ts_max_rank_corr_5'] / df['rank_stddev_adv5_10']
