# Calculate the rank of open prices
df['rank_open'] = df['open'].rank(axis=1, pct=True)
# Calculate the ts_argmin of the ranked open prices over the past 5 days
df['ts_argmin_rank_open_5'] = df['rank_open'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True)
# Calculate the correlation between high and volume over the past 10 days
df['correlation_high_volume_10'] = df['high'].rolling(window=10).corr(df['volume'])
# Calculate the rank of the product of the rank of volume over the past 5 days
df['rank_product_volume_5'] = df['volume'].rank(axis=1, pct=True).rolling(window=5).apply(lambda x: np.prod(x), raw=True).rank(axis=1, pct=True)
# Calculate the rank of the rank of the rank of open prices
df['rank_rank_rank_open'] = df['open'].rank(axis=1, pct=True).rank(axis=1, pct=True).rank(axis=1, pct=True)
# Calculate the alpha
df['alpha'] = ((-1 * df['ts_argmin_rank_open_5'].rank(axis=1, pct=True)) * df['correlation_high_volume_10'] + df['rank_product_volume_5'] * df['rank_rank_rank_open'])
