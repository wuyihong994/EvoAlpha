df['rank_stddev_open_10'] = df['open'].rolling(window=10).std().rank(axis=1, pct=True)
# Rank the trading volume over the past 5 days
df['ts_rank_volume_5'] = df['volume'].rolling(window=5).apply(lambda x: x.rank(pct=True).iloc[-1])
# Calculate the change in volume over the past 2 days
df['delta_volume_2'] = df['volume'] - df['volume'].shift(2)
# Calculate the condition for the mutation alpha
condition = ((-1 * df['rank_stddev_open_10']) * df['ts_rank_volume_5'] / df['delta_volume_2']) > df['high'].rolling(window=5).min()
# Sum the condition over the past 20 days
df['alpha'] = condition.rolling(window=20).sum()
