df['returns'] = df['close'].pct_change()
# Calculate the difference in returns between the current day and 4 days ago, then rank the results
df['delta_returns_4'] = df['returns'].diff(4)
df['rank_delta_returns_4'] = df['delta_returns_4'].rank(axis=1, pct=True)
# Calculate the 10-day average trading volume
df['adv10'] = df['volume'].rolling(window=10).mean()
# Calculate the sum of the 10-day average trading volume over the past 11 days, then find the maximum value over the past 5 days
df['sum_adv10_11'] = df['adv10'].rolling(window=11).sum()
df['ts_max_sum_adv10_11_5'] = df['sum_adv10_11'].rolling(window=5).max()
# Combine the ranked delta returns and the time-series max of the sum of the 10-day average trading volume to compute the alpha
df['alpha'] = -1 * df['rank_delta_returns_4'] / df['ts_max_sum_adv10_11_5']
