df['stddev_high_10'] = df['high'].rolling(window=10).std()
# Rank the standard deviation over the past 5 days
df['rank_stddev_high_10_5'] = df['stddev_high_10'].rolling(window=5).apply(lambda x: np.argmin(x), raw=True)
# Rank the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the correlation between high prices and ranked volume over the past 10 days
df['correlation_high_rank_volume_10'] = df['high'].rolling(window=10).corr(df['rank_volume'])
# Mutate Alpha
df['alpha'] = -1 * df['rank_stddev_high_10_5'] * df['correlation_high_rank_volume_10']
