df['sum_open_10'] = df['open'].rolling(window=10).sum()
# Rank the 10-day sum of opening prices
df['rank_sum_open_10'] = df['sum_open_10'].rank(axis=1, pct=True)
# Rank the volume
df['rank_volume'] = df['volume'].rank(axis=1, pct=True)
# Calculate the correlation between high and the ranked volume over the past 5 days
df['correlation_high_rank_volume_5'] = df['high'].rolling(window=5).corr(df['rank_volume'])
# Calculate the alpha factor
df['alpha'] = -1 * df['rank_sum_open_10'] * df['correlation_high_rank_volume_5']
